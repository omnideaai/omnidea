Hızlı Başlangıç
1. API Anahtarınızı Alın
# npm veya pip ile kurun
npm install @omnidea/sdk
# veya
pip install omnidea-python
2. İlk Tasarımı Üretin
import { OmnideaClient } from '@omnidea/sdk'

const client = new OmnideaClient({
  apiKey: 'omni_live_sk_...'
})

const design = await client.ai.generate({
  prompt: 'Modern berjer koltuk, velvet kumaş, altın bacaklar',
  sector: 'furniture',
  outputs: ['stl', 'dxf', 'pdf']
})

console.log(design.score) // Üretilebilirlik skoru: 88
console.log(design.files) // { stl: '...', dxf: '...', pdf: '...' }
3. Markete Gönderin
const listing = await client.market.publish({
  projectId: design.id,
  budget: { min: 800, max: 1500 },
  deadline: '2025-04-15',
  quantity: 1
})

// Webhook ile teklif bildirimlerini alın
client.webhooks.on('bid.received', (bid) => {
  console.log(`Yeni teklif: $${bid.amount} - ${bid.producer}`)
})
